#include <ESP32Servo.h>// Create Servo objectsServo servo1;Servo servo2;Servo servo3;Servo servo4;// EMG sensor pinconst int emgPin = 35;  // EMG sensor connected to GPIO34 (ADC1_CH6)// Servo motor pinsconst int servoPin1 = 13; //jjjjjjjjj Servo motor 1 connected to GPIO13const int servoPin2 = 14; // Servo motor 2 connected to GPIO14const int servoPin3 = 15; // Servo motor 3 connected to GPIO15const int servoPin4 = 12; // Servo motor 4 connected to GPIO16const int threshold = 778;  // Threshold for muscle contraction detectionvoid setup() {  // Initialize serial communication for debugging  Serial.begin(115200);    // Set ADC resolution to 12 bits  analogReadResolution(12);  analogSetAttenuation(ADC_11db);  // Set ADC attenuation for 0-3.3V range    // Attach servos to pins  servo1.attach(servoPin1);  servo2.attach(servoPin2);  servo3.attach(servoPin3);  servo4.attach(servoPin4);}void loop() {  int emgValue = analogRead(emgPin);  // Read EMG sensor value  // Print EMG value for debugging  Serial.print("EMG Value: ");  Serial.println(emgValue);  // Control servos based on EMG reading  if (emgValue <= threshold) {    // Rotate all servos to 0 degrees    servo1.write(0);    servo2.write(0);    servo3.write(0);    servo4.write(0);  } else {    // Rotate all servos to 90 degrees    servo1.write(90);    servo2.write(90);    servo3.write(90);    servo4.write(90);  }  delay(795);  // Small delay for stability}
